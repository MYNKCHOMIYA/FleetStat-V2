from fastapi import FastAPI
from app.routers import (
    users,
    trucks,
    trips,
    shipments,
    containers,
    customers,
    payments,
    fuel_logs,
    damage_reports,
    truck_services,
    trip_assignments,
    container_assignments,
    shipment_items,
    analytics
)

app = FastAPI(title="FleetStat API")

@app.get("/")
def root():
    return {"message": "FleetStat Backend Running"}

# Register all routers
app.include_router(users.router, tags=["users"])
app.include_router(trucks.router, tags=["trucks"])
app.include_router(trips.router, tags=["trips"])
# ... etc (all 14 routers registered)
